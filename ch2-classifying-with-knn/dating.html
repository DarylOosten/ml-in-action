euclidean = lambda candidate: lambda value: abs(candidate - value) ** 2
 
def classify(candidate):
    scores = normalized_data.apply(euclidean(candidate), axis=1).sum(axis = 1)
    scores = pd.concat([scores, data_like], axis = 1, keys = list(['score', 'like']))
    scores = scores.sort(columns=['score'])
    scores = scores.head(k)['like']
    return scores.value_counts().keys()[0]
